### Installation

<Steps>
  <Step>
    **Install Required Packages**

    ```package-install
    @radix-ui/react-accordion
    ```

  </Step>

  <Step>
    **Copy and paste the following code into  `mijn-ui` folder.**

    <Tabs items={["Source code", "Example usage"]}>
      <Tab value="Source code">
        ```json doc-gen:file
          {
            "file": "../../packages/mijn-ui/src/components/accordion/accordion.tsx",
            "codeblock": {
              "lang": "tsx",
              "meta": "title=\"mijn-ui/components/accordion.tsx\""
            }
          }
        ```
      </Tab>
      <Tab value="Example usage">
        ```json doc-gen:file
          {
            "file": "./examples/react/accordion/single-accordion.tsx",
            "codeblock": {
              "lang": "tsx",
              "meta": "title=\"single-accordion.tsx\""
            }
          }
        ```
      </Tab>
    </Tabs>

  </Step>

   <Step>
    **Add Animations to Your Tailwind Configuration**

```js title="tailwind.config.js"
/** @type {import('tailwindcss').Config} */
export default {
  theme: {
    extend: {
      keyframes: {
        // [!code highlight]
        "fade-in": {
          // [!code highlight]
          "0%": {
            // [!code highlight]
            opacity: "0", // [!code highlight]
          }, // [!code highlight]
          "100%": {
            // [!code highlight]
            opacity: "1", // [!code highlight]
          }, // [!code highlight]
        }, // [!code highlight]
        "fade-out": {
          // [!code highlight]
          "0%": {
            // [!code highlight]
            opacity: "1", // [!code highlight]
          }, // [!code highlight]
          "100%": {
            // [!code highlight]
            opacity: "0", // [!code highlight]
          }, // [!code highlight]
        }, // [!code highlight]
        "accordion-expand": {
          // [!code highlight]
          from: { height: "0" }, // [!code highlight]
          to: { height: "var(--radix-accordion-content-height)" }, // [!code highlight]
        }, // [!code highlight]
        "accordion-collapse": {
          // [!code highlight]
          from: {
            // [!code highlight]
            height: "var(--radix-accordion-content-height)", // [!code highlight]
          }, // [!code highlight]
          to: { height: "0" }, // [!code highlight]
        }, // [!code highlight]
      }, // [!code highlight]
      animation: {
        // [!code highlight]
        "accordion-expand": "accordion-expand 0.2s ease-in-out, fade-in 0.4s ease-in-out", // [!code highlight]
        "accordion-collapse": "accordion-collapse 0.2s ease-in-out, fade-out 0.4s ease-in-out", // [!code highlight]
      }, // [!code highlight]
    }, // [!code highlight]
  },
};
```

  </Step>
</Steps>
